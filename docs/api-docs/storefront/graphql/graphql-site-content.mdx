# Site Content with the GraphQL Storefront API

## Web pages

<Tabs items={['Request', 'Response']}>
<Tab>

```graphql filename="Example query: Get web pages" showLineNumbers copy
query {
  site {
    content {
      pages (filters: {isVisibleInNavigation: true }) {
        edges {
          node {
            __typename
            entityId
            name
            parentEntityId
            seo {
              pageTitle
              metaKeywords
              metaDescription
            }
            ... on NormalPage {
              plainTextSummary (characterLimit: 100)
              path
            }
            ... on ExternalLinkPage {
              link
            }
            ... on ContactPage {
              path
              plainTextSummary (characterLimit: 100)
              contactFields
            }
            ... on RawHtmlPage {
              path
              plainTextSummary (characterLimit: 100 )
            }
            ... on BlogIndexPage {
              path
            }
          }
        }
      }
    } 
  }
}
```

## Blogs

<Tabs items={['Request', 'Response']}>

query {
  site {
    content {
      blog {
        name
        description
        path
        posts (filters: { tags:["Most Popular", "Holiday"] } ) {
          edges {
            node {
              entityId
              name
              tags
              htmlBody
              plainTextSummary (characterLimit: 100)
              author
              path
              publishedDate {
                utc
              }
              thumbnailImage {
                url (width: 100, height: 200)
                urlOriginal
                altText
                isDefault
              }
              seo {
                pageTitle
                metaKeywords
                metaDescription
              }
            }
          }
        }
      }

  }
}
